Sentinel-2 Download via Copernicus Data Space
Search and download Sentinel-2 L2A scenes from the Copernicus Data Space Ecosystem using the cdsetool library.
Prerequisites
pip install cdsetool
from cdsetool.query import query_features from cdsetool.credentials import Credentials from cdsetool.download import download_features credentials = Credentials("your_username", "your_password") features = query_features( "Sentinel2", {"startDate": "2025-06-01", "completionDate": "2025-08-31", "processingLevel": "S2MSI2A", "box": [16.8, 46.6, 17.2, 46.9], # [west, south, east, north] "cloudCover": "[0,30]"} ) feature_list = list(features) print(f"Found {len(feature_list)} scenes") download_features(feature_list, "./sentinel2_data", credentials)
Expected Output
SAFE-format .zip files in ./sentinel2_data/, each ~800 MB containing 13 spectral bands as JP2 rasters at 10/20/60 m resolution.

Landsat Download via USGS M2M API
Search and download Landsat 8/9 Collection 2 Level-2 scenes using the landsatxplore library wrapping the USGS M2M API.
Prerequisites
pip install landsatxplore
from landsatxplore.api import API from landsatxplore.earthexplorer import EarthExplorer api = API("your_usgs_username", "your_usgs_password") scenes = api.search( dataset="landsat_ot_c2_l2", latitude=47.5, longitude=19.0, start_date="2025-06-01", end_date="2025-08-31", max_cloud_cover=20 ) print(f"Found {len(scenes)} Landsat scenes") api.logout() ee = EarthExplorer("your_usgs_username", "your_usgs_password") for scene in scenes[:3]: # download first 3 ee.download(scene["entity_id"], output_dir="./landsat_data") ee.logout()
Expected Output
Tar archives in ./landsat_data/, each ~200 MB containing 11 bands as GeoTIFF at 30 m (optical) and 100 m (thermal) resolution.

ERA5 Climate Reanalysis via CDS API
Download ERA5 hourly or monthly reanalysis data (temperature, precipitation, wind) from the Copernicus Climate Data Store.
Prerequisites
pip install cdsapi # Create ~/.cdsapirc with your CDS API key (https://cds.climate.copernicus.eu/)
import cdsapi client = cdsapi.Client() client.retrieve("reanalysis-era5-single-levels", { "product_type": "reanalysis", "variable": ["2m_temperature", "total_precipitation"], "year": "2024", "month": [f"{m:02d}" for m in range(1, 13)], "day": [f"{d:02d}" for d in range(1, 32)], "time": ["00:00", "06:00", "12:00", "18:00"], "area": [48, 16, 46, 23], # [N, W, S, E] — Hungary "format": "netcdf" }, "era5_hungary_2024.nc") print("Downloaded: era5_hungary_2024.nc")
Expected Output
NetCDF file (.nc) with gridded data at 0.25 degree resolution (~31 km). Typical size: 200-500 MB per variable per year for a country-scale area.

SRTM / Copernicus DEM via OpenTopography
Download the Copernicus GLO-30 DEM or SRTM 30m elevation data through the OpenTopography API with a free API key.
Prerequisites
pip install requests # Get free API key at https://opentopography.org/
import requests API_KEY = "your_opentopography_api_key" params = { "demtype": "COP30", # or "SRTMGL1" for SRTM 30m "south": 46.5, "north": 47.0, "west": 18.5, "east": 19.5, "outputFormat": "GTiff", "API_Key": API_KEY } r = requests.get("https://portal.opentopography.org/API/globaldem", params=params, stream=True) r.raise_for_status() with open("dem_budapest.tif", "wb") as f: for chunk in r.iter_content(chunk_size=8192): f.write(chunk) print(f"Downloaded: dem_budapest.tif ({r.headers.get('Content-Length', '?')} bytes)")
Expected Output
Single-band GeoTIFF at 30 m resolution. ~25 MB per 1 degree x 1 degree tile. Values are elevation in meters (WGS84 / EGM2008 geoid).

SoilGrids via WCS
Download global soil property maps (clay, sand, organic carbon, pH, etc.) from ISRIC SoilGrids at 250 m resolution using the WCS endpoint.
Prerequisites
pip install owslib rasterio
from owslib.wcs import WebCoverageService import rasterio from rasterio.io import MemoryFile wcs = WebCoverageService( "https://maps.isric.org/mapserv?map=/map/ocs.map", version="2.0.1") response = wcs.getCoverage( identifier=["ocs_0-30cm_mean"], # organic carbon stock 0-30 cm subsets=[("X", 18.5, 19.5), ("Y", 46.5, 47.0)], format="image/tiff" ) with MemoryFile(response.read()) as memfile: with memfile.open() as ds: data = ds.read(1) profile = ds.profile print(f"Shape: {data.shape}, CRS: {ds.crs}, Range: {data.min()}-{data.max()}") with open("soilgrids_ocs.tif", "wb") as f: response.seek(0) f.write(response.read())
Expected Output
GeoTIFF at 250 m resolution in Homolosine projection. Values are soil property values (units vary by layer). Available properties: clay, sand, silt, pH, SOC, bulk density, CEC, nitrogen.

OpenStreetMap Extract via Overpass API
Query and download OSM vector data (roads, buildings, land use) for a bounding box using the Overpass API, or download country PBF extracts from Geofabrik.
Prerequisites
pip install osmnx geopandas
import osmnx as ox import geopandas as gpd # Option 1: Download by place name buildings = ox.features_from_place("Budapest, Hungary", tags={"building": True}) print(f"Buildings: {len(buildings)}") buildings.to_file("budapest_buildings.gpkg", driver="GPKG") # Option 2: Download by bounding box roads = ox.features_from_bbox(bbox=(47.4, 47.6, 19.0, 19.2), tags={"highway": True}) roads.to_file("budapest_roads.gpkg", driver="GPKG") # Option 3: Geofabrik country extract (download PBF) import urllib.request url = "https://download.geofabrik.de/europe/hungary-latest.osm.pbf" urllib.request.urlretrieve(url, "hungary-latest.osm.pbf")
Expected Output
GeoPackage files with vector geometries (polygons for buildings, lines for roads). Geofabrik PBF is a compressed OSM extract (~200 MB for Hungary). CRS: WGS84 (EPSG:4326).

GBIF Species Occurrences via pygbif
Download species occurrence records from the Global Biodiversity Information Facility (GBIF) for a given taxon and geographic area.
Prerequisites
pip install pygbif pandas geopandas shapely
from pygbif import occurrences, species import pandas as pd import geopandas as gpd from shapely.geometry import Point # Find taxon key result = species.name_backbone(name="Bufo bufo", rank="species") taxon_key = result["usageKey"] # Download occurrences (Hungary bounding box) records = occurrences.search(taxonKey=taxon_key, country="HU", hasCoordinate=True, limit=300) df = pd.json_normalize(records["results"]) print(f"Records: {len(df)}") # Convert to GeoDataFrame geometry = [Point(r.decimalLongitude, r.decimalLatitude) for _, r in df.iterrows()] gdf = gpd.GeoDataFrame(df, geometry=geometry, crs="EPSG:4326") gdf.to_file("bufo_bufo_hungary.gpkg", driver="GPKG")
Expected Output
GeoPackage with point geometries. Each record includes species name, coordinates, date, institution, basis of record. CRS: WGS84 (EPSG:4326).

WorldPop Population Grid via API
Download high-resolution (100 m) gridded population estimates from WorldPop for any country and year.
Prerequisites
pip install requests rasterio
import requests, rasterio from rasterio.plot import show # WorldPop constrained individual countries UN-adjusted 100m iso3 = "HUN" year = 2020 url = (f"https://data.worldpop.org/GIS/Population/Global_2000_2020_Constrained/" f"2020/maxar_v1/{iso3}/{iso3.lower()}_ppp_{year}_UNadj_constrained.tif") print(f"Downloading WorldPop {iso3} {year}...") r = requests.get(url, stream=True) r.raise_for_status() fname = f"worldpop_{iso3}_{year}.tif" with open(fname, "wb") as f: for chunk in r.iter_content(chunk_size=8192): f.write(chunk) with rasterio.open(fname) as ds: print(f"Shape: {ds.shape}, CRS: {ds.crs}, Res: {ds.res}")
Expected Output
Single-band GeoTIFF at ~100 m resolution. Pixel values = estimated number of people. CRS: WGS84 (EPSG:4326). Typical file size: 50-200 MB per country.

GloFAS Flood Data via CDS API
Download Global Flood Awareness System (GloFAS) river discharge reanalysis or forecast data from the Copernicus Climate Data Store.
Prerequisites
pip install cdsapi xarray netCDF4 # Create ~/.cdsapirc with your CDS API key
import cdsapi import xarray as xr client = cdsapi.Client() client.retrieve("cems-glofas-historical", { "system_version": "version_4_0", "hydrological_model": "lisflood", "product_type": "consolidated", "variable": "river_discharge_in_the_last_24_hours", "hyear": "2023", "hmonth": ["january", "february", "march"], "hday": [f"{d:02d}" for d in range(1, 32)], "area": [48, 16, 46, 23], # [N, W, S, E] "format": "grib" }, "glofas_discharge_2023.grib") ds = xr.open_dataset("glofas_discharge_2023.grib", engine="cfgrib") print(ds)
Expected Output
GRIB or NetCDF file with daily river discharge values at ~0.05 degree (~5 km) resolution along river networks. Variable: dis24 in m3/s.

CORINE Land Cover via Copernicus CLMS
Download CORINE Land Cover raster or vector data from the Copernicus Land Monitoring Service. Covers all of Europe at 100 m resolution with 44 land cover classes.
Prerequisites
pip install requests zipfile-deflate64 rasterio geopandas
import requests, zipfile, os, rasterio from rasterio.mask import mask import geopandas as gpd from shapely.geometry import box # Download CORINE 2018 raster (100m) — requires Copernicus account # Direct download link (register at land.copernicus.eu first): url = ("https://land.copernicus.eu/en/products/corine-land-cover/" "clc2018?tab=download") print(f"Register and download manually from:\n{url}") # After download, clip to your AOI: aoi = gpd.GeoDataFrame(geometry=[box(18.5, 46.5, 19.5, 47.5)], crs="EPSG:4326").to_crs("EPSG:3035") with rasterio.open("U2018_CLC2018_V2020_20u1.tif") as src: clipped, transform = mask(src, aoi.geometry, crop=True) meta = src.meta.copy() meta.update({"height": clipped.shape[1], "width": clipped.shape[2], "transform": transform}) with rasterio.open("corine_clipped.tif", "w", **meta) as dst: dst.write(clipped) print("Clipped CORINE saved to corine_clipped.tif")
Expected Output
GeoTIFF raster at 100 m resolution in ETRS89/LAEA (EPSG:3035). Pixel values are CLC class codes (111-523). Full European raster is ~1.5 GB; clipped subsets are much smaller.
